using namespace std;

int main()
{
    string w, rev = "", rev1 = "";

    cout << "Enter a word: ";
    cin >> w;

    // process to get reverse
    for (int i = 0; i < w.length(); i++) // Traversing each letter
    {
        char ch = w.at(i); // Extract letter one by one
        rev = ch + rev;    // Reversing
        rev1 = "";         // 2nd reverse variable
        for (int j = 0; j < (w + rev).length() - 1; j++)
        {
            ch = (w + rev).at(j);
            rev1 = ch + rev1;
        }
        ch = (rev).at(rev.length() - 1);

        if (rev1 == w)
        {
            cout << "Palindrome of the word is " << w;
            break;
        }
        else if ((ch + rev1) == (w + rev))
        {
            cout << "Palindrome of the word " << w << " is " << rev;
            break;
        }
    }
    getch();
    return 0;
}
